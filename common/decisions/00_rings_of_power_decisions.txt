# Create the One Ring for your character
create_the_one_ring = {
	title = "Create the One Ring"
	major = yes
	# cooldown = { days = 30 }

	# Don't let AI create the ring
	ai_goal = no
	ai_check_interval = 0

	desc = "Create the One Ring for your own, with all the benefits - and risks - of doing so"

	# No other character may have created the ring
	is_valid_showing_failures_only = {
		NOT = {
			has_global_variable = global_one_ring_created
		}
	}

	# Hide if it cannot be created, since it can only be created once
	is_shown = {
		# One Ring has not been created
		NOT = {
			has_global_variable = global_one_ring_created
		}
	}

	effect = {
		add_trait = bearer_of_the_one_ring

		# Ensure it cannot be created again
		set_global_variable = {
			name = global_one_ring_created
			value = 1
		}
	}

	cost = {
		prestige = { 
			value = medium_prestige_value 
		}
	}

	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = 0
	}
}
