namespace = rop_events

# Remove One Ring from characters who die
rop_events.0001 = {
	type = character_event
	hidden = yes

	trigger = {
		has_trait = rop_bearer_of_one_ring
	}

	immediate = {
		# TODO Do not show notification
		remove_trait = rop_bearer_of_one_ring
	}
}

# Paranoid: stress gain from worrying about people taking the ring
# Character flags:
# 	rop_events.0002.has_triggered
rop_events.0002 = {
	type = character_event
	title = rop_events.0002.title
	desc = rop_events.0002.desc
	theme = mental_health

	left_portrait = {
		character = root
		animation = paranoia
	}

	trigger = {
		has_trait = rop_bearer_of_one_ring
		has_trait = paranoid
		NOT = {
			has_character_flag = rop_events.0002.has_triggered
		}
	}

	weight_multiplier = {
		base = 10
	}

	option = {
		name = rop_events.0002.option.a
		add_stress = 70
		add_character_flag = rop_events.0002.has_triggered
	}
}
